
import java.util.*;
public class Solution
{
	public static void main(String[] args)
	
	 {
		// write your code !!! 
		Scanner sc=new Scanner(System.in);
		byte b=sc.nextByte();
		for(byte i=1;i<=b;i++)
		{
			for(byte j=1;j<=i;j++)
			{
				System.out.print(i);
			}
			System.out.println();
		}
		   
	}
}
