import java.util.*;


public class Solution
{
	public static void main(String[] args) {
		 
		 Scanner sc = new Scanner(System.in);
		 int n=sc.nextInt();
		 for(int i=1;i<=n;i++)
		 {
			 char c=64;
			 for(int j=1;j<=i;j++)
			 {
				 c+=1;
				 System.out.print(c+" ");
			 }
			 System.out.println();
		 }
	}
}
